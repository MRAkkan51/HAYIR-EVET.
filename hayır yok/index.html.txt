    <script>
        let evetBoyut = 1;

        function sayfaGec(n) {
            document.getElementById('sayfa1').classList.add('hidden');
            document.getElementById('sayfa2').classList.add('hidden');
            document.getElementById('sayfa3').classList.add('hidden');
            document.getElementById('sayfa' + n).classList.remove('hidden');
        }

        function buyutEvet() {
            evetBoyut += 0.5; // Her hayırda %50 büyür
            const btnEvet = document.getElementById('btnEvet');
            btnEvet.style.transform = `scale(${evetBoyut})`;
            
            // Buton çok büyürse diğerlerinin üstüne binmemesi için z-index ekleyelim
            btnEvet.style.zIndex = "100";
        }
